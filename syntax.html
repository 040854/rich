1 from __future__ import annotations                                                                                                                                           
   2                                                                                                                                                                              
   3 import textwrap                                                                                                                                                              
   4 from typing import Any, Dict, Union                                                                                                                                          
   5                                                                                                                                                                              
   6 from pygments.lexers import get_lexer_by_name, guess_lexer_for_filename                                                                                                      
   7 from pygments.styles import get_style_by_name                                                                                                                                
   8 from pygments.token import Token                                                                                                                                             
   9 from pygments.util import ClassNotFound                                                                                                                                      
  10                                                                                                                                                                              
  11 from .color import Color, parse_rgb_hex, blend_rgb                                                                                                                           
  12 from .console import Console, ConsoleOptions, RenderResult, Segment, ConsoleRenderable                                                                                       
  13 from .style import Style                                                                                                                                                     
  14 from .text import Text                                                                                                                                                       
  15 from ._tools import iter_first                                                                                                                                               
  16                                                                                                                                                                              
  17                                                                                                                                                                              
  18 class Syntax:                                                                                                                                                                
  19     def __init__(                                                                                                                                                            
  20         self,                                                                                                                                                                
  21         code: str,                                                                                                                                                           
  22         lexer_name: str,                                                                                                                                                     
  23         *,                                                                                                                                                                   
  24         style: Union[str, Style] = None,                                                                                                                                     
  25         theme: str = "emacs",                                                                                                                                                
  26         dedent: bool = True,                                                                                                                                                 
  27         line_numbers: bool = False,                                                                                                                                          
  28         start_line: int = 1,                                                                                                                                                 
  29     ) -> None:                                                                                                                                                               
  30                                                                                                                                                                              
  31         if dedent:                                                                                                                                                           
  32             code = textwrap.dedent(code)                                                                                                                                     
  33         self.code = code                                                                                                                                                     
  34         self.lexer_name = lexer_name                                                                                                                                         
  35         self.style = style                                                                                                                                                   
  36         self.theme = theme                                                                                                                                                   
  37         self.dedent = dedent                                                                                                                                                 
  38         self.line_numbers = line_numbers                                                                                                                                     
  39         self.start_line = start_line                                                                                                                                         
  40                                                                                                                                                                              
  41         self._style_cache: Dict[Any, Style] = {}                                                                                                                             
  42                                                                                                                                                                              
  43         try:                                                                                                                                                                 
  44             self._pygments_style_class = get_style_by_name(theme)                                                                                                            
  45         except ClassNotFound:                                                                                                                                                
  46             self._pygments_style_class = get_style_by_name("default")                                                                                                        
  47                                                                                                                                                                              
  48         self._background_color = self._pygments_style_class.background_color                                                                                                 
  49                                                                                                                                                                              
  50     @classmethod                                                                                                                                                             
  51     def from_path(                                                                                                                                                           
  52         cls,                                                                                                                                                                 
  53         path: str,                                                                                                                                                           
  54         style: Union[str, Style] = None,                                                                                                                                     
  55         theme: str = "emacs",                                                                                                                                                
  56         dedent: bool = True,                                                                                                                                                 
  57         line_numbers: bool = False,                                                                                                                                          
  58         start_line: int = 1,                                                                                                                                                 
  59     ) -> Syntax:                                                                                                                                                             
  60         """Get a Syntax object for given path."""                                                                                                                            
  61         with open(path, "rt") as code_file:                                                                                                                                  
  62             code = code_file.read()                                                                                                                                          
  63         try:                                                                                                                                                                 
  64             lexer = guess_lexer_for_filename(path, code)                                                                                                                     
  65             lexer_name = lexer.name                                                                                                                                          
  66         except ClassNotFound:                                                                                                                                                
  67             lexer_name = "default"                                                                                                                                           
  68         return cls(                                                                                                                                                          
  69             code,                                                                                                                                                            
  70             lexer_name,                                                                                                                                                      
  71             style=style,                                                                                                                                                     
  72             theme=theme,                                                                                                                                                     
  73             dedent=dedent,                                                                                                                                                   
  74             line_numbers=line_numbers,                                                                                                                                       
  75             start_line=start_line,                                                                                                                                           
  76         )                                                                                                                                                                    
  77                                                                                                                                                                              
  78     def _get_theme_style(self, token_type) -> Style:                                                                                                                         
  79         if token_type in self._style_cache:                                                                                                                                  
  80             style = self._style_cache[token_type]                                                                                                                            
  81         else:                                                                                                                                                                
  82             pygments_style = self._pygments_style_class.style_for_token(token_type)                                                                                          
  83                                                                                                                                                                              
  84             color = pygments_style["color"]                                                                                                                                  
  85             bgcolor = pygments_style["bgcolor"]                                                                                                                              
  86             style = Style(                                                                                                                                                   
  87                 color="#" + color if color else "#000000",                                                                                                                   
  88                 bgcolor="#" + bgcolor if bgcolor else self._background_color,                                                                                                
  89                 bold=pygments_style["bold"],                                                                                                                                 
  90                 italic=pygments_style["italic"],                                                                                                                             
  91                 underline=pygments_style["underline"],                                                                                                                       
  92             )                                                                                                                                                                
  93             self._style_cache[token_type] = style                                                                                                                            
  94                                                                                                                                                                              
  95         return style.copy()                                                                                                                                                  
  96                                                                                                                                                                              
  97     def _get_default_style(self) -> Style:                                                                                                                                   
  98         style = self._get_theme_style(Token.Text)                                                                                                                            
  99         style = style.apply(Style(bgcolor=self._pygments_style_class.background_color))                                                                                      
 100         return style                                                                                                                                                         
 101                                                                                                                                                                              
 102     def _highlight(self, lexer_name: str) -> Text:                                                                                                                           
 103         default_style = self._get_default_style()                                                                                                                            
 104         try:                                                                                                                                                                 
 105             lexer = get_lexer_by_name(lexer_name)                                                                                                                            
 106         except ClassNotFound:                                                                                                                                                
 107             return Text(self.code, style=default_style)                                                                                                                      
 108         text = Text(style=default_style)                                                                                                                                     
 109         append = text.append                                                                                                                                                 
 110         _get_theme_style = self._get_theme_style                                                                                                                             
 111         for token_type, token in lexer.get_tokens(self.code):                                                                                                                
 112             append(token, _get_theme_style(token_type))                                                                                                                      
 113         return text                                                                                                                                                          
 114                                                                                                                                                                              
 115     def _get_line_numbers_color(self) -> Color:                                                                                                                              
 116         background_color = parse_rgb_hex(                                                                                                                                    
 117             self._pygments_style_class.background_color[1:]                                                                                                                  
 118         )                                                                                                                                                                    
 119         foreground_color = self._get_theme_style(Token.Text)._color                                                                                                          
 120         if foreground_color is None:                                                                                                                                         
 121             return Color.default()                                                                                                                                           
 122         # TODO: Handle no full colors here                                                                                                                                   
 123         assert foreground_color.triplet is not None                                                                                                                          
 124         new_color = blend_rgb(background_color, foreground_color.triplet)                                                                                                    
 125         return Color.from_triplet(new_color)                                                                                                                                 
 126                                                                                                                                                                              
 127     def __console__(self, console: Console, options: ConsoleOptions) -> RenderResult:                                                                                        
 128         code = self.code                                                                                                                                                     
 129         if self.dedent:                                                                                                                                                      
 130             code = textwrap.dedent(code)                                                                                                                                     
 131         text = self._highlight(self.lexer_name)                                                                                                                              
 132         if not self.line_numbers:                                                                                                                                            
 133             yield text                                                                                                                                                       
 134             return                                                                                                                                                           
 135                                                                                                                                                                              
 136         lines = text.split("\n")                                                                                                                                             
 137         numbers_column_width = len(str(self.start_line + len(lines))) + 2                                                                                                    
 138         render_options = options.with_width(options.max_width - numbers_column_width)                                                                                        
 139                                                                                                                                                                              
 140         if self.style is None:                                                                                                                                               
 141             background_style = Style(                                                                                                                                        
 142                 bgcolor=self._pygments_style_class.background_color                                                                                                          
 143             )                                                                                                                                                                
 144         else:                                                                                                                                                                
 145             if isinstance(self.style, str):                                                                                                                                  
 146                 background_style = console.get_style(self.style) or Style()                                                                                                  
 147             else:                                                                                                                                                            
 148                 background_style = self.style                                                                                                                                
 149                                                                                                                                                                              
 150         number_style = background_style.apply(self._get_theme_style(Token.Text))                                                                                             
 151         number_style._color = self._get_line_numbers_color()                                                                                                                 
 152         padding = Segment(" " * numbers_column_width, background_style)                                                                                                      
 153         new_line = Segment("\n")                                                                                                                                             
 154         for line_no, line in enumerate(lines, self.start_line):                                                                                                              
 155             wrapped_lines = console.render_lines(line, render_options)                                                                                                       
 156             for first, wrapped_line in iter_first(wrapped_lines):                                                                                                            
 157                 if first:                                                                                                                                                    
 158                     yield Segment(                                                                                                                                           
 159                         f" {str(line_no).rjust(numbers_column_width - 2)} ",                                                                                                 
 160                         number_style,                                                                                                                                        
 161                     )                                                                                                                                                        
 162                 else:                                                                                                                                                        
 163                     yield padding                                                                                                                                            
 164                 yield from wrapped_line                                                                                                                                      
 165                 yield new_line
